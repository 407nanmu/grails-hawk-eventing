001    package grails.plugins.hawkeventing;
002    
003    import groovy.lang.Closure;
004    
005    import java.util.Collections;
006    import java.util.HashMap;
007    import java.util.HashSet;
008    import java.util.Map;
009    import java.util.Set;
010    
011    /**
012     * 
013     * @author Kim A. Betti
014     */
015    public class EventBroker {
016    
017        private Map<String, Set<EventConsumer>> eventConsumers = Collections.synchronizedMap(new HashMap<String, Set<EventConsumer>>());
018    
019        private EventPublisher eventPublisher;
020    
021        public void subscribe(Set<EventSubscription> subscriptions) {
022            for (EventSubscription subscription : subscriptions) {
023                subscribe(subscription);
024            }
025        }
026    
027        public void subscribe(String eventName, Closure eventClosure) {
028            ClosureEventConsumer consumer = new ClosureEventConsumer(eventClosure);
029            subscribe(eventName, consumer);
030        }
031    
032        public void subscribe(EventSubscription subscription) {
033            String eventName = subscription.getEventName();
034            subscribe(eventName, subscription.getConsumer());
035        }
036    
037        public void subscribe(String eventName, EventConsumer consumer) {
038            ensureKeyFor(eventName);
039            eventConsumers.get(eventName).add(consumer);
040        }
041    
042        private void ensureKeyFor(String eventName) {
043            if (!eventConsumers.containsKey(eventName)) {
044                eventConsumers.put(eventName, new HashSet<EventConsumer>());
045            }
046        }
047    
048        public void unsubscribe(String eventName, EventConsumer eventConsumer) {
049            Set<EventConsumer> consumers = getEventConsumers(eventName);
050            consumers.remove(eventConsumer);
051        }
052    
053        /**
054         * Use this method to publish a basic event with any kind
055         * of payload. The payload will be wrapped in a BaseEvent
056         * before published to any subscribers.
057         * @param eventName
058         * @param payload
059         */
060        public void publish(String eventName, Object payload) {
061            Event event = new BaseEvent(eventName, payload);
062            publishEvent(event);
063        }
064    
065        public void publishEvent(Event event) {
066            String fullEventName = event.getEventName();
067            EventNameDecoder eventNameDecoder = new EventNameDecoder(fullEventName);
068    
069            while (eventNameDecoder.hasNext()) {
070                String currentEventName = eventNameDecoder.next();
071                Set<EventConsumer> consumers = getEventConsumers(currentEventName);
072                for (EventConsumer consumer : consumers) {
073                    eventPublisher.publish(event, consumer);
074                }
075            }
076        }
077    
078        private Set<EventConsumer> getEventConsumers(String eventName) {
079            Set<EventConsumer> consumers = Collections.emptySet();
080            if (eventConsumers.containsKey(eventName)) {
081                consumers = eventConsumers.get(eventName);
082            }
083    
084            return consumers;
085        }
086    
087        public void setEventPublisher(EventPublisher publisher) {
088            this.eventPublisher = publisher;
089        }
090    
091    }
