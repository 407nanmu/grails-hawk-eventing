001    package grails.plugins.hawkeventing.annotation;
002    
003    import java.lang.annotation.ElementType;
004    import java.lang.annotation.Retention;
005    import java.lang.annotation.RetentionPolicy;
006    import java.lang.annotation.Target;
007    
008    /**
009     * Annotate the class containing tenant information with this. This annotation
010     * should only be applied to one domain class per project.
011     */
012    @Retention(RetentionPolicy.RUNTIME)
013    @Target(ElementType.METHOD)
014    public @interface Consuming {
015        String[] value();
016    }
