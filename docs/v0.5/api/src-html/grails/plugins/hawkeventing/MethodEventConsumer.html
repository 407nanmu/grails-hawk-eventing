001    package grails.plugins.hawkeventing;
002    
003    import grails.plugins.hawkeventing.exceptions.EventException;
004    
005    import java.lang.reflect.InvocationTargetException;
006    import java.lang.reflect.Method;
007    
008    /**
009     * Support publishing of events to specific method.
010     * This is most likely configured by annotations.
011     * 
012     * Note! It makes use of the reflection API so there
013     * might be a slight performance trade of involved
014     * with using this.
015     * 
016     * @author Kim A. Betti
017     */
018    public class MethodEventConsumer implements EventConsumer {
019    
020        private final Method method;
021        private final Object consumer;
022    
023        public MethodEventConsumer(Object consumer, Method method) {
024            this.consumer = consumer;
025            this.method = method;
026        }
027    
028        @Override
029        public void consume(Event event) {
030            try {
031                method.invoke(consumer, event);
032            } catch (IllegalAccessException ex) {
033                onReflectionException(ex, event);
034            } catch (IllegalArgumentException ex) {
035                onReflectionException(ex, event);
036            } catch (InvocationTargetException ex) {
037                onReflectionException(ex, event);
038            }
039        }
040    
041        private void onReflectionException(Exception ex, Event event) {
042            String exMessage = "Unable to notify " + consumer + " of "
043            + event.getEventName() + " because of reflection related exception";
044            throw new EventException(exMessage, ex);
045        }
046    
047    }
